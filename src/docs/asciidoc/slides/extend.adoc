== Extension example

.`+issue:<id>[]+`
[source.condense]
----
asciidoctor {
  extensions {
    inline_macro (name: 'issue') { parent, id, attrs ->
      issues_uri = parent.getAttr('uri-issues')
      options = [
        type: ':link',
        target: "${issues_uri}/${id}".toString()
      ]
      createInline(parent, 'anchor',
          target, attrs, options).render()
    }
  }
}
----
